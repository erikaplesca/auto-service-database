-- CREAREA DE TABELE

CREATE TABLE CLIENT ( 
	id_client NUMBER PRIMARY KEY,    
	adresa VARCHAR2(100) NOT NULL,    
	telefon VARCHAR2(15) NOT NULL,    
	email VARCHAR2(50)
);

CREATE TABLE PERSOANA_FIZICA (    
	id_client NUMBER PRIMARY KEY,    
	nume VARCHAR2(30) NOT NULL,    
	prenume VARCHAR2(30) NOT NULL,    
	cnp CHAR(13) UNIQUE NOT NULL,    
	CONSTRAINT subentitate_pf_client FOREIGN KEY (id_client)         
		REFERENCES CLIENT(id_client),    
	CONSTRAINT marime_cnp CHECK (LENGTH(cnp) = 13)
);
	

CREATE TABLE PERSOANA_JURIDICA (    
	id_client NUMBER PRIMARY KEY,    
	denumire VARCHAR2(50) NOT NULL,    
	cui CHAR(8) UNIQUE NOT NULL,    
	CONSTRAINT subentitate_pj_client FOREIGN KEY (id_client)        
		REFERENCES CLIENT(id_client),    
	CONSTRAINT marime_cui CHECK (LENGTH(cui) = 8)
);

CREATE TABLE MASINA (   
	id_masina NUMBER PRIMARY KEY,    
	id_client NUMBER NOT NULL,    
	tip_masina VARCHAR2(20) NOT NULL,    
	marca VARCHAR2(20) NOT NULL,    
	model VARCHAR2(20) NOT NULL,    
	an_fabricatie NUMBER(4) NOT NULL,    
	nr_inmatriculare VARCHAR2(10) UNIQUE NOT NULL,    
	CONSTRAINT fk_masina_client FOREIGN KEY (id_client)        
	REFERENCES CLIENT(id_client)
);

CREATE TABLE PROGRAMARE (    
	id_programare NUMBER PRIMARY KEY,    
	id_client NUMBER NOT NULL,    
	id_masina NUMBER NOT NULL,    
	data_ora TIMESTAMP NOT NULL,    
	CONSTRAINT fk_prog_client FOREIGN KEY (id_client)        
		REFERENCES CLIENT(id_client),    
	CONSTRAINT fk_prog_masina FOREIGN KEY (id_masina)        
		REFERENCES MASINA(id_masina)
);

CREATE TABLE DEVIZ (    
	id_deviz NUMBER PRIMARY KEY,    
	id_programare NUMBER UNIQUE NOT NULL,    
	data_eliberarii DATE NOT NULL,    	
	cost_estimat NUMBER(10,2) NOT NULL,   
	status VARCHAR2(15) DEFAULT 'In elaborare'         
	CHECK (status IN ('In elaborare', 'Trimis', 'Acceptat', 'Respins')),   
	CONSTRAINT fk_deviz_prog FOREIGN KEY (id_programare)
		REFERENCES PROGRAMARE(id_programare)
);

CREATE TABLE SERVICIU (
	id_serviciu NUMBER PRIMARY KEY NOT NULL,
	nume_serviciu VARCHAR2(50) NOT NULL,
	tip_masina VARCHAR2(20) NOT NULL
	CHECK (tip_masina in('Turism','SUV','Autoutilitara')),
	cost NUMBER(7,2) NOT NULL CHECK (cost > 0)
);

CREATE TABLE SERVICIU_DEVIZ (
	id_deviz NUMBER NOT NULL,
	id_serviciu NUMBER NOT NULL,
	PRIMARY KEY (id_deviz, id_serviciu),    
	CONSTRAINT fk_sd_deviz FOREIGN KEY (id_deviz)
		REFERENCES DEVIZ(id_deviz),
	CONSTRAINT fk_sd_serviciu FOREIGN KEY (id_serviciu)
		REFERENCES SERVICIU(id_serviciu)
);

CREATE TABLE ANGAJATI (
	id_angajat NUMBER PRIMARY KEY NOT NULL,
	nume VARCHAR2(30) NOT NULL,
	prenume VARCHAR2(30) NOT NULL,
	salariu NUMBER(7,2) NOT NULL CHECK(salariu >=0)
);

CREATE TABLE MECANIC (
	id_angajat NUMBER PRIMARY KEY NOT NULL,
	tip_mecanic VARCHAR2(20) NOT NULL
	CHECK (tip_mecanic IN     
	('Mecanic sef', 'Mecanic general', 'Specialist motor', 'Ajutor mecanic')),    
	CONSTRAINT fk_mecanic_ang FOREIGN KEY (id_angajat)    
		REFERENCES ANGAJATI(id_angajat));

CREATE TABLE CONTABIL (
    id_angajat NUMBER PRIMARY KEY NOT NULL,
    poate_emite_facturi CHAR(1) NOT NULL 
    CHECK (poate_emite_facturi IN ('T', 'F')),
    tip_contabil VARCHAR2(20),
    CONSTRAINT fk_contabil_ang FOREIGN KEY (id_angajat)
    REFERENCES ANGAJATI(id_angajat)
);

CREATE TABLE FACTURA (
    id_factura NUMBER PRIMARY KEY,
    id_client NUMBER NOT NULL,
    id_masina NUMBER NOT NULL,
    id_angajat NUMBER NOT NULL,
    cost_total NUMBER(10,2) NOT NULL CHECK (cost_total >= 0),
    data_emitere DATE NOT NULL,
    data_scadenta DATE NOT NULL,
    status VARCHAR2(10) NOT NULL CHECK (status IN ('Platita', 'Neplatita')),
    CONSTRAINT fk_factura_client FOREIGN KEY (id_client)
        REFERENCES CLIENT(id_client),
    CONSTRAINT fk_factura_masina FOREIGN KEY (id_masina)
        REFERENCES MASINA(id_masina),
    CONSTRAINT fk_factura_angajat FOREIGN KEY (id_angajat)
        REFERENCES CONTABIL(id_angajat)
);

CREATE TABLE PROCES_REP (
    id_proces NUMBER PRIMARY KEY,
    id_factura NUMBER, 
    id_deviz NUMBER NOT NULL,
    CONSTRAINT fk_proces_factura FOREIGN KEY (id_factura)
        REFERENCES FACTURA(id_factura)
        ON DELETE SET NULL,
    CONSTRAINT fk_proces_deviz FOREIGN KEY (id_deviz)
        REFERENCES DEVIZ(id_deviz)
);

CREATE TABLE SERVICIU_PROCES (
    id_proces NUMBER NOT NULL,
    id_serviciu NUMBER NOT NULL,
    data_incepere DATE,
    data_finalizare DATE,
    status VARCHAR2(20) NOT NULL
        CHECK (status IN ('Neinceput', 'In lucru', 'Finalizat')),
    PRIMARY KEY (id_proces, id_serviciu),
    CONSTRAINT fk_sp_proces FOREIGN KEY (id_proces)
        REFERENCES PROCES_REP(id_proces),
    CONSTRAINT fk_sp_serviciu FOREIGN KEY (id_serviciu)
        REFERENCES SERVICIU(id_serviciu)
);

CREATE TABLE ANGAJATI_RESPONSABILI (
    id_angajat NUMBER NOT NULL,
    id_proces NUMBER NOT NULL,
    id_serviciu NUMBER NOT NULL,
    PRIMARY KEY (id_angajat, id_proces, id_serviciu),
    CONSTRAINT fk_ar_angajat FOREIGN KEY (id_angajat)
        REFERENCES MECANIC(id_angajat),
    CONSTRAINT fk_ar_proces_serviciu FOREIGN KEY (id_proces, id_serviciu)
        REFERENCES SERVICIU_PROCES(id_proces, id_serviciu)
);

CREATE TABLE FURNIZOR (
    id_furnizor NUMBER PRIMARY KEY,
    nume VARCHAR2(50) NOT NULL,
    procent_reducere NUMBER(3,2) DEFAULT 0.00
);

CREATE TABLE PIESA (
    id_piesa NUMBER PRIMARY KEY NOT NULL,
    id_furnizor NUMBER NOT NULL,
    denumire VARCHAR2(50) NOT NULL,
    pret_standard NUMBER(7,2) NOT NULL,
    FOREIGN KEY (id_furnizor) REFERENCES FURNIZOR(id_furnizor)
);

CREATE TABLE PIESE_UTILIZATE (
    id_proces NUMBER NOT NULL,
    id_serviciu NUMBER NOT NULL,
    id_piesa NUMBER NOT NULL,
    data_comanda DATE,
    data_livrare DATE,
    PRIMARY KEY (id_proces, id_serviciu, id_piesa),
    CONSTRAINT fk_pu_serviciu_proces FOREIGN KEY (id_proces, id_serviciu)
        REFERENCES SERVICIU_PROCES(id_proces, id_serviciu),
    CONSTRAINT fk_pu_piesa FOREIGN KEY (id_piesa)
        REFERENCES PIESA(id_piesa)
);

        'Ulei motor 5W40', 'Filtru ulei', 'Filtru aer'

    (s.nume_serviciu LIKE 'Inlocuit refrigerant%' AND p.denumire IN (
        'Refrigerant R134A'
    ))
    OR
    (s.nume_serviciu = 'Manopera electric' AND p.denumire IN (
        'Baterie auto', 'Bec far'
    ))
)
WHERE NOT EXISTS (
    SELECT 1 FROM PIESE_UTILIZATE pu
    WHERE pu.id_proces = sp.id_proces
      AND pu.id_serviciu = sp.id_serviciu
      AND pu.id_piesa = p.id_piesa
);

---- ANGAJATI_RESPONSABILI
-- asociem automat mecanicii la servicii_proces, echitabil, alfabetic (round-robin)
INSERT INTO ANGAJATI_RESPONSABILI (id_angajat, id_proces, id_serviciu)
WITH
--1.identificăm pentru fiecare serviciu_proces, tippul de mecanic necesar
SP_TIP AS (
  SELECT
    sp.id_proces,
    sp.id_serviciu,
    CASE
      WHEN s.nume_serviciu LIKE 'Inlocuit ulei si filtre%' THEN 'Mecanic general'
      WHEN s.nume_serviciu LIKE 'Manopera accesorii motor%' THEN 'Mecanic general'
      WHEN s.nume_serviciu LIKE 'Reglat geometrie roti%' THEN 'Mecanic general'
      WHEN s.nume_serviciu LIKE 'Test franare%' THEN 'Mecanic general'
      WHEN s.nume_serviciu LIKE 'Inlocuit refrigerant%' THEN 'Specialist motor'
      WHEN s.nume_serviciu LIKE 'Diagnoza computerizata%' THEN 'Mecanic sef'
    END AS tip_mecanic
  FROM SERVICIU_PROCES sp
  JOIN SERVICIU s ON sp.id_serviciu = s.id_serviciu
),

-- 2. listez mecnicii disponibili pentru fiecare tip, ordonaTi alfabetic
MECANICI_ORD AS (
  SELECT
    m.id_angajat,
    m.tip_mecanic,
    a.nume,
    a.prenume,
    ROW_NUMBER() OVER (PARTITION BY m.tip_mecanic ORDER BY a.nume, a.prenume) AS rn,
    COUNT(*) OVER (PARTITION BY m.tip_mecanic) AS total_mec
  FROM MECANIC m
  JOIN ANGAJATI a ON m.id_angajat = a.id_angajat
),

-- 3.asociem fiecarui serviciu_proces un index( pentru round-robin)
SP_INDEX AS (
  SELECT
    st.*,
    ROW_NUMBER() OVER (PARTITION BY st.tip_mecanic ORDER BY st.id_proces, st.id_serviciu) AS idx
  FROM SP_TIP st
)

-- 4.alocam mecanicul corespunzător fiecărui serviciu_proces
--fac atribuirea pe rad: primul serviciu ia primul mecanic, 
-- al doilea serviciu ia al doilea mecanic etc 
-- iar dacă sunt mai multe servicii decat mecanici,
-- se reia de la început (round-robin).

--asa, toti mecanicii dintr-o categorie vor primi 
-- un număr cât mai apropiat de lucrari, fara ca vreunul să fie supraincărcat.


SELECT
  mo.id_angajat,
  si.id_proces,
  si.id_serviciu
FROM SP_INDEX si
JOIN MECANICI_ORD mo
  ON mo.tip_mecanic = si.tip_mecanic
  AND mo.rn = MOD(si.idx - 1, mo.total_mec) + 1
WHERE NOT EXISTS (
    SELECT 1 FROM ANGAJATI_RESPONSABILI ar
    WHERE ar.id_proces = si.id_proces AND ar.id_serviciu = si.id_serviciu
);



-- Vreau ca toate procesele sa aiba statusul 'Finalizat', ca sa le pot genera facturi.
UPDATE SERVICIU_PROCES sp
SET status = 'Finalizat',
    data_incepere = (
        SELECT d.data_eliberarii
        FROM PROCES_REP p
        JOIN DEVIZ d ON p.id_deviz = d.id_deviz
        WHERE p.id_proces = sp.id_proces
    ),
    data_finalizare = (
        SELECT d.data_eliberarii + INTERVAL '2' DAY
        FROM PROCES_REP p
        JOIN DEVIZ d ON p.id_deviz = d.id_deviz
        WHERE p.id_proces = sp.id_proces
    );

----- FACTURA
INSERT INTO FACTURA (
    id_factura,
    id_client,
    id_masina,
    id_angajat,
    cost_total,
    data_emitere,
    data_scadenta,
    status
)
SELECT
   SEQ_FACTURA.NEXTVAL,
    c.id_client,
    m.id_masina,
    (SELECT ca.id_angajat
       FROM CONTABIL ca
      WHERE ca.poate_emite_facturi = 'T'
        AND ROWNUM = 1),
    NVL(
        (SELECT SUM(s.cost)
           FROM SERVICIU_PROCES sp
           JOIN SERVICIU s ON sp.id_serviciu = s.id_serviciu
          WHERE sp.id_proces = pr.id_proces),
        0
    )
    +
    NVL(
        (SELECT SUM(p.pret_standard * (1 - NVL(f.procent_reducere, 0)))
           FROM PIESE_UTILIZATE pu
           JOIN PIESA p ON pu.id_piesa = p.id_piesa
           JOIN FURNIZOR f ON p.id_furnizor = f.id_furnizor
          WHERE pu.id_proces = pr.id_proces),
        0
    ),
    SYSDATE,
    CASE
        WHEN pf.id_client IS NOT NULL THEN SYSDATE + 7
        WHEN pj.id_client IS NOT NULL THEN SYSDATE + 30
        ELSE SYSDATE + 7
    END,
    'Neplatita'
FROM PROCES_REP pr
JOIN DEVIZ d ON pr.id_deviz = d.id_deviz
JOIN PROGRAMARE p ON d.id_programare = p.id_programare
JOIN MASINA m ON p.id_masina = m.id_masina
JOIN CLIENT c ON m.id_client = c.id_client
LEFT JOIN PERSOANA_FIZICA pf ON c.id_client = pf.id_client
LEFT JOIN PERSOANA_JURIDICA pj ON c.id_client = pj.id_client
WHERE NOT EXISTS (
    SELECT 1
      FROM SERVICIU_PROCES sp
     WHERE sp.id_proces = pr.id_proces
       AND sp.status <> 'Finalizat'
);
